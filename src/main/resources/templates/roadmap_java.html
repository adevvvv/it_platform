    Lambda
    Без параметров: 
    @FunctionalInterface
interface MyName{
    // абстрактный метод
    double getPiValue();
}

public class Main {
    public static void main( String[] args ) {

    // объявление ссылки на функциональный интерфейс MyName
    MyName pi;
// --------------------
    // лямбда-выражение:
    pi = () -> 3.1415;
// --------------------
    System.out.println("Число Пи = " + pi.getPiValue()); // Вывод: Число Пи = 3.1415
    }
}
    С параметрами: 
    @FunctionalInterface
interface WordReverse {
    String reverse(String n);
}

public class Main {
    public static void main( String[] args ) {

        WordReverse word;

        word = (str) -> {

            StringBuilder result = new StringBuilder();
            for (int i = str.length()-1; i >= 0 ; i--)
                result.append(str.charAt(i));
            return result.toString();
        };
        // вызов метода из интерфейса
        System.out.println("Перевернутое слово = " + word.reverse("Lambda")); // Вывод: Перевернутое слово = adbmaL
    }

}
